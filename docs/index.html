fbx – Console Framebuffer Library

Version 1.0.0 – Cross‑platform terminal graphics & input in C++17

▸ Overview

fbx gives you a framebuffer for the terminal: a 2D grid of cells where each cell holds a UTF‑8 grapheme, foreground/background colors (256‑palette or truecolor), and text styles (bold, italic, underline, etc.). You can draw with escape‑sequence‑aware printing, composite multiple buffers, manage layers or tiled layouts, read keys with full modifier support, and use familiar std::ostream/std::istream interfaces.

Works on Linux (via termios, poll) and Windows (via virtual terminal sequences). Requires C++17.

▸ Core features

  • Unicode + ANSI cell model (grapheme clusters, wide characters, combining)
  • Dirty‑tracking and differential rendering
  • Region snapshots (save/restore any rectangular area)
  • Layer compositing (z‑order, offset, transparency via empty cell)
  • Tiling (horizontal/vertical split with proportional weights)
  • Raw keyboard input with Ctrl, Alt, Shift modifiers
  • fbxostream – write to a framebuffer like std::cout
  • fbxistream – read a line with inline editing inside the framebuffer
  • Escape sequence parser (cursor movement, SGR colors, clear screen, etc.)

▸ Framebuffer (fbx::Framebuffer)

Main drawing surface.

Construction & dimensions

Framebuffer fb(80, 24);
fb.resize(120, 30, true);   // preserve content
fb.fit_to_terminal();       // resize to current terminal size

Cell access

fb.put_char(col, row, "█", Color::yellow(), Color::blue(), Style_Bold);
Cell& c = fb.cell(x, y);
std::string ch = fb.get_char(x, y);

High‑level printing (escape‑sequence aware)

fb.set_cursor(5, 10);
fb.print("\x1b[1;31mHello \x1b[4mWorld\x1b[0m!\n");
// supports: \n, \r, \t, ESC[A/B/C/D, ESC[H, ESC[m, ESC[2J, ESC[K, etc.

Region snapshots

Rect area = {10, 5, 20, 10};
Snapshot snap = fb.save_region(area);
// ... draw something over the area ...
fb.restore_region(snap);  // revert

Compositing

Framebuffer overlay(10, 5);
overlay.put_char(0, 0, "X", Color::red());
fb.composite(overlay, 5, 5);  // paste at offset, skipping transparent cells

Rendering

fb.render();           // incremental, only dirty cells
fb.render_region({0,0,10,10});
std::string ansi = fb.to_ansi();  // full image as ANSI string

▸ FramebufferManager

Manages multiple framebuffers and composes them into a single terminal output.

Buffer lifecycle

FramebufferManager mgr;
FbId bg = mgr.create(80, 24, "background");
FbId fg = mgr.create_fullscreen("overlay");
Framebuffer* fb = mgr.get(bg);

Layer mode (compositing with z‑order)

mgr.add_layer(bg, 0, 0, 0);  // bottom
mgr.add_layer(fg, 5, 3, 1);  // top
mgr.set_layer_visible(fg, true);
mgr.flush();  // composite and render

Tile mode (split screen)

TileLayout layout;
layout.dir = TileDir::Horizontal;
layout.panes = { {left_id, 1}, {right_id, 2} }; // weights 1 and 2
mgr.set_tiles(layout);
mgr.flush();

Active buffer (single fullscreen)

mgr.set_active(fb_id);
mgr.flush();

Terminal resize handling

void on_sigwinch(int) { mgr.on_resize(); }

▸ Stream helpers (fbx::fbxostream, fbx::fbxistream)

Convenient std::ostream/std::istream wrappers around a framebuffer.

Output stream

Framebuffer fb(80, 24);
fbxostream out(fb);
out << fbx::fg(Color::red()) << "Error: " << fbx::reset << message << '\n';
out.at(10, 5) << "Positioned text";

Input stream (line editor inside framebuffer)

fbxistream in(fb);
std::string name = in.readline("Name: ");  // shows prompt, handles backspace, echo

▸ Complete examples

Example 1: Key inspector (raw mode)

#include <fbx/fbx.hpp>
int main() {
    fbx::console::enter_alt_screen();
    fbx::console::hide_cursor();
    fbx::console::set_raw_mode(true);

    fbx::Framebuffer fb(fbx::console::get_size());
    fb.print("Press any key (Ctrl+Q to quit)");

    while (true) {
        auto k = fbx::console::getch();
        if (k.is_ctrl('q') || k.is_escape()) break;

        fb.clear();
        fb.print("Key: " + k.name + "\n");
        fb.print("Raw: ");
        for (unsigned char c : k.raw)
            fb.print(fbx::fmt::hex(c) + " ");
        fb.render();
    }

    fbx::console::set_raw_mode(false);
    fbx::console::show_cursor();
    fbx::console::leave_alt_screen();
}

Example 2: Layer compositing with snapshot

#include <fbx/fbx.hpp>
int main() {
    using namespace fbx;
    console::enter_alt_screen();
    console::set_raw_mode(true);

    FramebufferManager mgr;
    FbId bg = mgr.create_fullscreen("bg");
    FbId dlg = mgr.create(40, 10, "dialog");
    mgr.add_layer(bg, 0, 0, 0);
    mgr.add_layer(dlg, 5, 3, 1);

    Framebuffer* bgfb = mgr.get(bg);
    // draw checkerboard background
    for (int r = 0; r < bgfb->rows(); ++r)
        for (int c = 0; c < bgfb->cols(); ++c)
            bgfb->put_char(c, r, ((r+c)%2) ? "▒" : "░");

    Snapshot behind = bgfb->save_region({5,3,40,10});

    Framebuffer* dlgfb = mgr.get(dlg);
    dlgfb->fill_rect({0,0,40,10}, Cell{" ", Color::none(), Color::blue()});
    dlgfb->print("\x1b[1;37mDialog on top\x1b[0m");

    mgr.flush();
    console::read_key(-1); // wait for key

    bgfb->restore_region(behind);
    mgr.remove_layer(dlg);
    mgr.flush();

    console::set_raw_mode(false);
    console::leave_alt_screen();
}

Example 3: Using fbxostream and fbxistream

#include <fbx/fbx.hpp>
int main() {
    using namespace fbx;
    console::enter_alt_screen();
    console::set_raw_mode(true);

    Framebuffer fb(console::get_size());
    fbxostream out(fb);
    fbxistream in(fb);

    out << "\x1b[1;36mStream demo\x1b[0m\n";
    out << "Enter your name: ";
    fb.render();  // show prompt

    std::string name = in.readline();
    out << "\nHello, " << name << "!\nPress any key to exit.";
    fb.render();
    console::read_key(-1);

    console::set_raw_mode(false);
    console::leave_alt_screen();
    return 0;
}

▸ Building & integration

fbx uses CMake. Include it via add_subdirectory or find_package.

# In your CMakeLists.txt:
add_subdirectory(path/to/fbx)
target_link_libraries(your_app PRIVATE fbx::fbx)

You can also use CPM like this:

cmake_minimum_required(VERSION 3.15)
project(MyFbxProject)

set(CMAKE_CXX_STANDARD 17)

# ---- Add CPM.cmake ----
if(NOT EXISTS ${CMAKE_CURRENT_BINARY_DIR}/cmake/CPM.cmake)
    file(DOWNLOAD https://github.com/cpm-cmake/CPM.cmake/releases/latest/download/get_cpm.cmake
         ${CMAKE_CURRENT_BINARY_DIR}/cmake/CPM.cmake)
endif()
include(${CMAKE_CURRENT_BINARY_DIR}/cmake/CPM.cmake)

# ---- Declare fbx dependency ----
CPMAddPackage(
    NAME fbx
    GITHUB_REPOSITORY eLiteEs/fbx 
    GIT_TAG 1.0.0
)

# ---- Your application ----
add_executable(my_app main.cpp)
target_link_libraries(my_app PRIVATE fbx::fbx)

Compile with C++17 and enable terminal features on Windows (link with winmm and user32 automatically).

Examples can be built with -DFBX_BUILD_EXAMPLES=ON.
